       ```typescript
       import type { FxProvider } from '@budget/shared-kernel';
       import type { FxRateCacheRepo } from '../../ports/fx-rate-cache-repo';
       import { formatDateUTC } from './format-date-utc';

       export class NoFxRateAvailable extends Error {
         constructor(public from: string, public to: string, public date: string) {
           super(`No FX rate available for ${from}/${to} as of ${date}`);
           this.name = 'NoFxRateAvailable';
         }
       }

       export class FrankfurterFxProvider implements FxProvider {
         constructor(
           private readonly cache: FxRateCacheRepo,
           private readonly fetchFn: typeof fetch = fetch,
         ) {}

         async rateAsOf(from: string, to: string, date: Date) {
           if (from === to) return { rate: '1', provider: 'frankfurter', isStale: false };
           const yyyymmdd = formatDateUTC(date);
           const cached = await this.cache.lookup(from, to, yyyymmdd);
           if (cached) return { rate: cached.rate, provider: 'frankfurter', isStale: cached.date !== yyyymmdd };
           try {
             const r = await this.fetchFn(`https://api.frankfurter.dev/v2/rate/${from}/${to}?date=${yyyymmdd}`);
             if (!r.ok) throw new Error(`frankfurter http ${r.status}`);
             const j = (await r.json()) as { date: string; rate: number };
             const rateStr = String(j.rate);  // ACL: number → string at boundary
             await this.cache.upsert(from, to, j.date, rateStr, 'frankfurter');
             return { rate: rateStr, provider: 'frankfurter', isStale: j.date !== yyyymmdd };
           } catch {
             const fallback = await this.cache.mostRecentPrior(from, to, yyyymmdd);
             if (!fallback) throw new NoFxRateAvailable(from, to, yyyymmdd);
             return { rate: fallback.rate, provider: 'frankfurter', isStale: true };
           }
         }
       }
       ```

       import type { PgBoss } from 'pg-boss';
       import { withInfraTx } from '@budget/platform';
       import { sql } from 'drizzle-orm';
       import type { FxProvider } from '@budget/shared-kernel';

       export function registerFxDailyFetch(boss: PgBoss, fxProvider: FxProvider) {
         boss.work('fx-daily-fetch', async () => {
           const pairs = await withInfraTx(async (tx) => {
             const rows = await tx.execute(sql`
               SELECT DISTINCT currency_orig AS base, currency_default AS quote FROM budgeting.expense_ledger
               UNION
               SELECT DISTINCT currency_orig AS base, default_currency AS quote
                 FROM budgeting.expense_ledger e
                 JOIN tenancy.workspaces w ON w.id = e.tenant_id
             `);
             return rows.rows as Array<{ base: string; quote: string }>;
           });
           const today = new Date();
           let ok = 0, fail = 0;
           for (const { base, quote } of pairs) {
             try { await fxProvider.rateAsOf(base, quote, today); ok++; }
             catch { fail++; }
           }
           return { fetched: ok, failed: fail };
         });
       }
