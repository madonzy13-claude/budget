<objective>
Ship the `Idempotency-Key` HTTP middleware that protects every mutating route in Phase 2 (transactions, accounts, categories, etc.) from PWA offline-then-reconnect duplicate submission. Storage in `shared_kernel.idempotency_keys` (D-05-c — supersedes the original `platform.idempotency_keys` mention). Two-layer dedup: scope_hash (tenant + user + route + key) for lookup; body_hash for body-mismatch detection (Stripe pattern). 24h TTL with hourly pg-boss cleanup. Race-safe via SELECT FOR UPDATE inside withTenantTx.

Purpose: EXPN-12 mandates `Idempotency-Key` on every mutating endpoint. The middleware is plan-02 deliverable so plans 02-04 (accounts), 02-05 (categories), and 02-06 (transactions) can register their routes already protected. Without this plan, those plans would have to retroactively wrap their routes.

Output: A registered Hono middleware running AFTER `tenantGuard` and BEFORE route handlers; a pg-boss cleanup job; integration tests covering replay, body-hash mismatch, TTL expiry, scope-leak prevention, body-survives-middleware, and race conditions.
</objective>

       ```typescript
       import { sql } from 'drizzle-orm';
       import type { Tx } from '../db/tx';

       export interface IdempotencyRow {
         scopeHash: string; bodyHash: string; responseStatus: number;
         responseBodyJsonb: unknown; expiresAt: Date;
       }

       export async function lookupIdempotency(tx: Tx, scopeHash: string): Promise<IdempotencyRow | null> {
         const r = await tx.execute<IdempotencyRow>(sql`
           SELECT scope_hash AS "scopeHash", body_hash AS "bodyHash",
                  response_status AS "responseStatus", response_body_jsonb AS "responseBodyJsonb",
                  expires_at AS "expiresAt"
             FROM shared_kernel.idempotency_keys
            WHERE scope_hash = ${scopeHash}
              AND expires_at > now()
            FOR UPDATE
         `);
         return r.rows[0] ?? null;
       }

       export async function insertIdempotency(tx: Tx, row: {
         scopeHash: string; bodyHash: string; tenantId: string; userId: string;
         route: string; responseStatus: number; responseBody: unknown;
       }): Promise<void> {
         await tx.execute(sql`
           INSERT INTO shared_kernel.idempotency_keys
             (scope_hash, body_hash, tenant_id, user_id, route, response_status, response_body_jsonb, expires_at)
           VALUES (${row.scopeHash}, ${row.bodyHash}, ${row.tenantId}, ${row.userId}, ${row.route},
                   ${row.responseStatus}, ${JSON.stringify(row.responseBody)}::jsonb,
                   now() + interval '24 hours')
         `);
       }

       export async function deleteExpiredIdempotency(tx: Tx): Promise<number> {
         const r = await tx.execute(sql`DELETE FROM shared_kernel.idempotency_keys WHERE expires_at < now()`);
         return r.rowCount ?? 0;
       }
       ```

       ```typescript
       import { createHash } from 'node:crypto';
       import type { Context, Next } from 'hono';
       import { withTenantTx } from '../db/tx';
       import { lookupIdempotency, insertIdempotency } from './repo';
       import { TenantId, UserId } from '@budget/shared-kernel';

       const MUTATING = new Set(['POST', 'PUT', 'PATCH', 'DELETE']);

       function sha256(s: string): string {
         return createHash('sha256').update(s).digest('hex');
       }

       export interface IdempotencyContext {
         tenantId: string; userId: string;
       }

       export function createIdempotencyMiddleware<E extends { Variables: { tenantId?: string; userId?: string } }>() {
         return async (c: Context<E>, next: Next) => {
           if (!MUTATING.has(c.req.method)) return next();
           const key = c.req.header('Idempotency-Key');
           if (!key) return next();
           const tenantId = c.get('tenantId');
           const userId = c.get('userId');
           if (!tenantId || !userId) return next();  // pre-auth route; skip

           const route = c.req.path;
           const scopeHash = sha256(`${tenantId}|${userId}|${route}|${key}`);

           // LOCKED body-survival strategy: clone the underlying Web Request and read text.
           // c.req.raw is the underlying Request; cloning it leaves the original body intact for
           // downstream zValidator('json', ...) → c.req.json() consumers (Hono v4.12+).
           const rawBody = c.req.method === 'GET' ? '' : await c.req.raw.clone().text();
           const bodyHash = sha256(rawBody);

           const r = await withTenantTx(TenantId(tenantId), UserId(userId), async (tx) => {
             const cached = await lookupIdempotency(tx, scopeHash);
             if (cached) {
               if (cached.bodyHash !== bodyHash) {
                 return { kind: 'mismatch' as const };
               }
               return { kind: 'replay' as const, status: cached.responseStatus, body: cached.responseBodyJsonb };
             }
             return { kind: 'fresh' as const };
           });

           if (r.isErr()) throw r.error;

           if (r.value.kind === 'mismatch') {
             return c.json({ error: 'idempotency_key_reused_with_different_body' }, 422);
           }
           if (r.value.kind === 'replay') {
             return c.json(r.value.body as object, r.value.status as 200);
           }

           // Fresh — body has NOT been consumed (we only cloned). Hand control to downstream handler.
           // Optional fallback hook: also stash the raw body on context so handlers that need raw text can read it
           // without re-buffering the request stream.
           c.set('idempotency_raw_body' as any, rawBody);

           await next();

           // Capture response and persist
           const status = c.res.status;
           const respText = await c.res.clone().text();
           let body: unknown;
           try { body = JSON.parse(respText); } catch { body = respText; }

           await withTenantTx(TenantId(tenantId), UserId(userId), async (tx) => {
             await insertIdempotency(tx, {
               scopeHash, bodyHash, tenantId, userId, route,
               responseStatus: status, responseBody: body,
             });
           });
         };
       }
       ```
