// create-share-link.ts
import { nanoid } from "nanoid";
export async function createShareLink(
  deps,
  { budgetId, tenantId, userId, ttlDays = 7 },
) {
  // assertOwnerRole (Better Auth) — see existing pattern in tenancy/src/application/invite-member.ts
  const token = nanoid(32);
  const expiresAt = new Date(Date.now() + ttlDays * 24 * 60 * 60 * 1000);
  const link = await deps.repo.create({
    budgetId,
    tenantId,
    createdBy: userId,
    token,
    expiresAt,
  });
  return {
    url: `${process.env.APP_URL}/budgets/join/${token}`,
    expiresAt,
    id: link.id,
  };
}

// resolve-share-link.ts (PUBLIC — no auth, no tenant context)
export async function resolveShareLink(deps, token: string) {
  const link = await deps.repo.findByToken(token);
  if (!link) return { found: false };
  const now = new Date();
  const budget = await deps.budgetRepo.findByIdNoTenant(link.budgetId); // also infra-level
  return {
    found: true,
    budgetName: budget?.name ?? "(unknown)",
    isExpired: link.expiresAt < now,
    isRevoked: link.revokedAt !== null,
    isUsed: link.acceptedBy !== null,
  };
}

// accept-share-link.ts
export async function acceptShareLink(
  deps,
  token: string,
  acceptingUserId: string,
) {
  const link = await deps.repo.findByToken(token);
  if (!link) throw new Error("NotFound");
  const now = new Date();
  if (link.revokedAt !== null) throw new Error("Revoked");
  if (link.expiresAt < now) throw new Error("Expired");
  if (link.acceptedBy !== null) throw new Error("AlreadyUsed");

  // Better Auth addMember — per RESEARCH §Code Examples
  await deps.auth.api.addMember({
    body: {
      organizationId: link.budgetId,
      userId: acceptingUserId,
      role: "member",
    },
  });

  await deps.repo.accept(link.id, acceptingUserId);
  return { budgetId: link.budgetId };
}

// revoke-share-link.ts
export async function revokeShareLink(deps, { linkId, tenantId, userId }) {
  // assertOwnerRole
  await deps.repo.revoke(linkId, tenantId);
}

import { Hono } from "hono";
import { zValidator } from "@hono/zod-validator";
import { z } from "zod";

export function createShareJoinRoute(deps) {
  const app = new Hono();

  // PUBLIC: no requireAuth, no requireWorkspace — called by recipient before login
  app.get("/:token", async (c) => {
    const token = c.req.param("token");
    const result = await resolveShareLink(deps, token);
    if (!result.found) return c.json({ error: "NotFound" }, 404);
    return c.json(result);
  });

  // AUTHENTICATED: requireAuth applied via app.use in app.ts; needs session but NOT active budget
  app.post("/:token/accept", async (c) => {
    const userId = c.var.userId; // from authMiddleware
    if (!userId) return c.json({ error: "Unauthenticated" }, 401);
    const token = c.req.param("token");
    try {
      const { budgetId } = await acceptShareLink(deps, token, userId);
      return c.json({ budgetId }, 200);
    } catch (e: any) {
      if (e.message === "NotFound") return c.json({ error: "NotFound" }, 404);
      if (e.message === "Revoked") return c.json({ error: "Revoked" }, 410);
      if (e.message === "Expired") return c.json({ error: "Expired" }, 410);
      if (e.message === "AlreadyUsed")
        return c.json({ error: "AlreadyUsed" }, 409);
      throw e;
    }
  });

  return app;
}
