```typescript
// One ordered event in a category's (or budget's) history. Amounts are integer cents.
export type ReserveEngineEvent =
  | {
      type: "setLimit";
      categoryId: string;
      month: string;
      normalCents: bigint;
      cushionCents: bigint;
    } // SCD-2 effective limit for a month
  | {
      type: "spendDelta";
      categoryId: string;
      month: string;
      deltaCents: bigint;
    } // +add txn / −remove txn / net edit
  | { type: "adjust"; categoryId: string; deltaCents: bigint } // signed reserve adjustment (op 3 via delta)
  | { type: "accrual"; categoryId: string; month: string } // month-close: reserve += left (op 4)
  | { type: "cushion"; month: string; on: boolean } // budget-level cushion mode change (SCD-2)
  | { type: "exclude"; categoryId: string; excluded: boolean } // reserve_excluded toggle (drops R from internal)
  | { type: "archive"; categoryId: string; mode: "all" | "current_future" } // category deletion (decision J)
  | { type: "setUserDefined"; cents: bigint }; // Σ RESERVE-wallet balances (surplus input only)

export interface CategoryReserveState {
  reserveCents: bigint; // R — available reserve carried forward chronologically
  usedCents: bigint; // U — reserve already consumed by overspend
  // capacity = R + U is implicit (only changes on adjust/accrual/exclude/delete)
}

// Per-(category, month) display row.
export interface CategoryMonthCell {
  categoryId: string;
  month: string; // 'YYYY-MM'
  overageCents: bigint; // max(spent − effLimit, 0)
  leftCents: bigint; // max(effLimit − spent, 0)
  usedCents: bigint; // reserve drawn for THIS month
  overspentCents: bigint; // overage − used  (per-month)
}

export interface ReserveEngineResult {
  // Final running R/U per category after replaying ALL events.
  states: Map<string, CategoryReserveState>;
  // Per-(category,month) cells for display.
  cells: CategoryMonthCell[];
  internalCents: bigint; // Σ R over active (non-excluded, non-archived) categories
  userDefinedCents: bigint; // last setUserDefined value (default 0n)
  surplusCents: bigint; // userDefined − internal
}

export function reserveEngine(opts: {
  events: ReserveEngineEvent[];
  // 'YYYY-MM' of the open month. Accrual applies only to CLOSED months (< openMonth).
  openMonth: string;
  reservesEnabled?: boolean; // default true. false → see decision K transform.
}): ReserveEngineResult;
```

Implement `reserveEngine` as a chronological fold. COPY THESE FORMULAS VERBATIM from 05-REWRITE-SPEC.md (lines 30-68) — do NOT paraphrase:

PER-CATEGORY STATE (spec lines 30-39):

```
R = available reserve; U = used reserve; capacity = R + U
effLimit = cushionModeOn ? category.cushionLimit : category.normalLimit
overage   = max(spent − effLimit, 0)      // spent = Σ this category's txns in the month
left      = max(effLimit − spent, 0)
overspent = overage − U                    // hard invariant: U + overspent = overage
```

THE 4 OPERATIONS (spec lines 43-60) — these mutate R/U as the running `overage` for a (category) changes:

```
1. overage increases by Δ:
       draw = min(Δ, R);  R -= draw;  U += draw          // remainder (Δ − draw) becomes overspent
2. overage decreases by Δ:
       dec = Δ
       fromOverspent = min(dec, overspent)               // cut overspent first
       remaining     = dec − fromOverspent
       U -= remaining;  R += remaining                    // return used → available
3. set reserve to X (X ≥ 0):
       d = X − R
       if d >= 0:  cover = min(d, overspent);  U += cover;  R += (d − cover)   // cover overspent first
       else:       R += d                                                       // (R becomes X)
4. month-close accrual:  reserve += left   // applied exactly as op 3 with X = R + left
```

GLOBALS (spec lines 64-67):

```
internal = Σ over active (non-excluded, non-archived) categories of R
surplus  = userDefined − internal
```
