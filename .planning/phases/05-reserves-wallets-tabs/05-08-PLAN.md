    import { Page, Locator, expect } from "@playwright/test";

    export class WalletsPage {
      constructor(private page: Page) {}

      async open(budgetName: string) {

        // OR resolve via the BudgetSwitcher: depends on Phase 3 navigation pattern

      section(type: "SPENDINGS" | "CUSHION" | "RESERVE"): Locator {
        return this.page.getByTestId(`wallet-section-${type}`);
      }
      sectionHeader(type: "SPENDINGS" | "CUSHION" | "RESERVE"): Locator {
        return this.section(type).getByRole("heading");
      }

      // W-5: row by UUID via testid template (used after we resolve id by name)
      row(walletId: string): Locator {
        return this.page.locator(`[data-wallet-id="${walletId}"]`);
      }

      // W-5: locate a persisted row by visible name (Name cell text); returns the row element.
      // Plan 05 emits data-testid="wallet-row" (semantic) AND data-wallet-id (UUID) on every persisted row.
      rowByName(name: string): Locator {
        return this.page.locator('[data-testid="wallet-row"]', { hasText: name });
      }

      // W-5: resolve UUID from visible name. Returns the wallet's id as a string.
      async resolveIdByName(name: string): Promise<string> {
        const row = this.rowByName(name);
        await row.waitFor({ state: "visible" });
        const id = await row.getAttribute("data-wallet-id");
        if (!id) throw new Error(`Wallet "${name}" found but data-wallet-id is empty (is it a draft? W-5 contract: draft rows have empty data-wallet-id)`);
        return id;
      }

      // Draft row (W-4 staged-add) — exposed for the add-staged feature scenarios
      draftRow(): Locator { return this.page.locator('[data-testid="wallet-row-draft"]'); }
      draftNameInput(): Locator { return this.page.locator('[data-testid="wallet-draft-name-input"]'); }

      addButton(type: "spendings" | "cushion" | "reserve"): Locator {
        return this.page.getByTestId(`add-wallet-${type}`);
      }
      nameCell(walletId: string): Locator { return this.page.getByTestId(`wallet-name-${walletId}`); }
      currencyCell(walletId: string): Locator { return this.page.getByTestId(`wallet-currency-${walletId}`); }
      amountCell(walletId: string): Locator { return this.page.getByTestId(`wallet-amount-${walletId}`); }
      trashButton(walletId: string): Locator { return this.page.getByTestId(`wallet-trash-${walletId}`); }

      async editName(walletId: string, newName: string) {
        await this.nameCell(walletId).click();
        const editor = this.page.getByTestId(`wallet-name-${walletId}-editor`).locator("input");
        await editor.fill(newName);
        await editor.blur();
      }
      async editAmount(walletId: string, newAmount: string) {
        await this.amountCell(walletId).click();
        const editor = this.page.getByTestId(`wallet-amount-${walletId}-editor`).locator("input");
        await editor.fill(newAmount);
        await editor.blur();
      }
      async editCurrency(walletId: string, newCurrency: string) {
        await this.currencyCell(walletId).click();
        // Currency picker shape depends on Plan 05 implementation — adjust selectors
        await this.page.getByRole("option", { name: newCurrency }).click();
      }
      async dragToSection(walletId: string, targetType: "SPENDINGS" | "CUSHION" | "RESERVE") {
        const handle = this.row(walletId).getByRole("button", { name: /Drag to move/ });
        const target = this.section(targetType);
        await handle.hover();
        await this.page.mouse.down();
        await target.hover();
        await this.page.mouse.up();

      async deleteWallet(walletId: string, confirm = true) {

        await this.trashButton(walletId).click();

        await expect(dialog).toBeVisible();
        if (confirm) await dialog.getByRole("button", { name: /Delete/ }).click();
        else await dialog.getByRole("button", { name: /Cancel/ }).click();

      // W-4 staged-add helper
      async addWalletStaged(type: "spendings" | "cushion" | "reserve", name: string): Promise<string> {
        await this.addButton(type).click();
        await this.draftRow().waitFor({ state: "visible" });
        await this.draftNameInput().fill(name);
        await this.draftNameInput().blur();
        // After commit the draft disappears and a persisted row with `name` appears.
        return this.resolveIdByName(name);
      }

      toast(text: string | RegExp): Locator {
        return this.page.locator('[data-sonner-toast]', { hasText: text });
      }

    // Wallets steps

      async function (this: any, budgetName, name, type, currency, amount) {
        // Use the API (POST /wallets with the fresh-user cookie session) to exercise more of the stack —

        const res = await this.apiContext.post('/wallets', {
          data: { name, walletType: type, currency, amount },

        });
        if (!res.ok()) throw new Error(`seed wallet failed: ${res.status()} ${await res.text()}`);

    When('I open the Wallets tab on a budget {string}', async function (this: any, _budgetName: string) {
      const wallets = new WalletsPage(this.page);
      await wallets.open(this.budgetId);
    });

    When('I click {string}', async function (this: any, label: string) {
      // Generic click; for "Add spendings wallet" routes to the section-typed add button
      const m = label.match(/^Add (spendings|cushion|reserve) wallet$/i);
      if (m) {
        const wallets = new WalletsPage(this.page);
        await wallets.addButton(m[1].toLowerCase() as any).click();
        return;
      }
      await this.page.getByRole('button', { name: label }).click();
    });

      // Otherwise resolve persisted UUID via W-5 data-wallet-id and use the InlineEditCell flow.
      const draft = wallets.draftRow();
      if (await draft.count() > 0 && await draft.locator(`[data-testid="wallet-draft-name-input"]`).inputValue() === "") {
        await wallets.draftNameInput().fill(newName);
        await wallets.draftNameInput().blur();
        return;

      const id = await wallets.resolveIdByName(currentName);
      await wallets.editName(id, newName);
    });
