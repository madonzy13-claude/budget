```typescript
export interface ReservePosition {
  categoryId: string;
  reserveCents: bigint;       // R (available reserve shown to user)
  usedCents: bigint;          // U (reserve consumed by overspend, cumulative)
  overspentCents: bigint;     // ╬Ż per-month overspent for this category (= overage ŌłÆ U summed)
  // per-month projection for the spendings grid: 'YYYY-MM' ŌåÆ { used, overspent, overage, left }
  byMonth: Map<string, { usedCents: bigint; overspentCents: bigint; overageCents: bigint; leftCents: bigint }>;
}

export interface ReservePositionsResult {
positions: Map<string, ReservePosition>;
internalCents: bigint; // ╬Ż R over active (non-excluded, non-archived) categories
userDefinedCents: bigint; // ╬Ż RESERVE-wallet balances
surplusCents: bigint; // userDefined ŌłÆ internal
direction: "TOPUP" | "WITHDRAW" | "NONE"; // surplus<0 ŌåÆ TOPUP, surplus>0 ŌåÆ WITHDRAW, 0 ŌåÆ NONE
}

````
NOTE direction mapping (matches existing recompute-reserve-topup-task sign convention): `surplus = userDefined ŌłÆ internal`; `surplus < 0 ŌåÆ TOPUP` (internal>userDefined: need to top up the wallet), `surplus > 0 ŌåÆ WITHDRAW`.
