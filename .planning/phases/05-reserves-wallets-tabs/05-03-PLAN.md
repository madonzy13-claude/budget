    ```typescript
    import { ok, err, type Result } from "@budget/shared-kernel";
    import { Money } from "@budget/shared-kernel";
    import type { WalletRepo } from "../ports/wallet-repo";
    import type { WalletType } from "../domain/wallet";

    export interface UpdateWalletDeps {
      repo: WalletRepo;
      budgetCurrencyOf: (tenantId: string) => Promise<string>;
    }

    export function updateWallet(deps: UpdateWalletDeps) {
      return async (input: {
        tenantId: string;
        walletId: string;
        actorUserId: string;
        name?: string;
        amount?: string;          // numeric string from Zod
        currency?: string;
        walletType?: WalletType;
      }): Promise<Result<{ wallet: { id: string; name: string; walletType: WalletType; currency: string; currentBalanceCents: string } }, Error>> => {
        try {
          const wallet = await deps.repo.findById(input.tenantId, input.walletId);
          if (!wallet) return err(new Error("not_found"));

          const effectiveType = input.walletType ?? wallet.walletType;
          const effectiveCurrency = input.currency ?? wallet.currency;

          if (effectiveType === "RESERVE") {
            const budgetCcy = await deps.budgetCurrencyOf(input.tenantId);
            if (effectiveCurrency !== budgetCcy) {
              return err(new Error("reserve_currency_mismatch"));
            }
          }

          // Apply domain mutators in order; abort on first err
          if (input.name !== undefined) {
            const r = wallet.rename(input.name);
            if (r.isErr()) return err(r.error);
          }
          if (input.walletType !== undefined) {
            const r = wallet.changeType(input.walletType);
            if (r.isErr()) return err(r.error);
          }
          if (input.currency !== undefined) {
            const r = wallet.changeCurrency(input.currency);
            if (r.isErr()) return err(r.error);
          }
          if (input.amount !== undefined) {
            // Amount must be expressed in wallet's POST-change currency
            const amt = Money.of(input.amount, wallet.currency);
            const r = wallet.setAmount(amt);
            if (r.isErr()) return err(r.error);
          }

          await deps.repo.update(input.tenantId, input.walletId, {
            name: input.name,
            amount: input.amount,
            currency: input.currency ? input.currency.toUpperCase() : undefined,
            walletType: input.walletType,
          }, input.actorUserId);

          return ok({ wallet: {
            id: wallet.id, name: wallet.name, walletType: wallet.walletType,
            currency: wallet.currency,
            currentBalanceCents: (Number(wallet.currentBalance.amount) * 100).toFixed(0),
          }});
        } catch (e) { return err(e as Error); }
      };
    }
    ```

    export function getReservesSummary(deps: GetReservesSummaryDeps) {
      return async (input: { tenantId: string; budgetId: string }): Promise<Result<ReservesSummaryDto, Error>> => {
        try {
          const enabled = await deps.isReservesEnabled(input.tenantId);
          const budgetCurrency = await deps.budgetCurrencyOf(input.tenantId);

          if (!enabled) {
            return ok({
              rows: [],
              excludedRows: [],
              totals: { totalCategoryReservesCents: "0", totalReserveWalletAmountCents: "0", mismatchCents: "0", disabled: true, budgetCurrency },
            });
          }

          const [activeBalanceMap, excludedBalanceMap, categories, reserveWalletSum] = await Promise.all([
            deps.reserveBalanceRepo.getForBudget(input.budgetId, input.tenantId, new Date()),
            deps.reserveBalanceRepo.getExcludedForBudget(input.budgetId, input.tenantId, new Date()),
            deps.categoriesRepo.list(input.tenantId),
            deps.reservesSummaryRepo.sumReserveWalletAmounts(input.tenantId),
          ]);

          // Partition categories by reserveExcluded
          const activeCats = (categories as any[]).filter((c) => !c.reserveExcluded);
          const excludedCats = (categories as any[]).filter((c) => c.reserveExcluded);

          // Active row balances (drives totals + share math)
          let totalCategoryReserves = 0n;
          const activeRowBalances: { id: string; name: string; cents: bigint }[] = [];
          for (const c of activeCats) {
            const m = activeBalanceMap.get(c.id);
            const cents = m ? BigInt(Math.round(Number(m.amount) * 100)) : 0n;
            activeRowBalances.push({ id: c.id, name: c.name, cents });
            totalCategoryReserves += cents;
          }
          const totalWallets = reserveWalletSum;
          const rows: ReservesSummaryRow[] = activeRowBalances.map((r) => {
            const sharePct = (totalCategoryReserves === 0n || totalWallets === 0n)
              ? null
              : Number((r.cents * 10000n) / totalCategoryReserves) / 100; // 2 decimal precision
            const shareAmt = sharePct === null ? null
              : ((r.cents * totalWallets) / (totalCategoryReserves === 0n ? 1n : totalCategoryReserves)).toString();
            return {
              categoryId: r.id, name: r.name,
              reserveBalanceCents: r.cents.toString(),
              walletSharePercent: sharePct,
              walletShareAmountCents: shareAmt,
            };
          });

          // Excluded rows — REAL FROZEN balance from getExcludedForBudget map (W-3 fix)
          // Share is always null for Excluded; they're informational only.
          // NEVER synthesize "0" as a placeholder — only emit "0" when the map genuinely
          // has no entry (category truly has zero auto-compute + zero adjustments).
          const excludedRows: ReservesSummaryRow[] = excludedCats.map((c) => {
            const m = excludedBalanceMap.get(c.id);
            const cents = m ? BigInt(Math.round(Number(m.amount) * 100)) : 0n;
            return {
              categoryId: c.id, name: c.name,
              reserveBalanceCents: cents.toString(),
              walletSharePercent: null,
              walletShareAmountCents: null,
            };
          });

          return ok({
            rows,
            excludedRows,
            totals: {
              totalCategoryReservesCents: totalCategoryReserves.toString(),
              totalReserveWalletAmountCents: totalWallets.toString(),
              mismatchCents: (totalWallets - totalCategoryReserves).toString(),
              disabled: false,
              budgetCurrency,
            },
          });
        } catch (e) { return err(e as Error); }
      };
    }
