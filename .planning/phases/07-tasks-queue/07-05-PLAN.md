     * Recompute reserve mismatch and emit-or-resolve the RESERVE_TOPUP task.
     * MUST be called inside an existing withTenantTx — `tx` is mandatory.
     * Idempotent: safe to call repeatedly.
     */
    export async function recomputeReserveTopupTask(
      tx: TenantTx,
      input: RecomputeReserveTopupTaskInput,
      deps: RecomputeReserveTopupTaskDeps,
    ): Promise<void> {
      // 1. Compute current summary via the shared builder (single source of math per D-PH7-03).
      const summary = await buildReservesSummary({
        tx,
        tenantId: input.tenantId,
        budgetId: input.budgetId,
        // Pass the same deps the builder expects — adjust signature per actual builder shape.
        categoriesRepo: deps.categoriesRepo,
        reserveBalanceRepo: deps.reserveBalanceRepo,
        reservesSummaryRepo: deps.reservesSummaryRepo,
        budgetCurrencyOf: deps.budgetCurrencyOf,
      });

      // 2. Extract mismatch (bigint via string round-trip — DTO carries string)
      const mismatchCents = BigInt(summary.totals.mismatchCents);

      if (mismatchCents === 0n) {
        await deps.taskRepo.resolveByKindAndBudget(
          input.tenantId,
          input.budgetId,
          "RESERVE_TOPUP",
          tx,
        );
        return;
      }

      // 3. Emit (idempotent via ON CONFLICT DO NOTHING on partial unique index)
      const absShortfall = mismatchCents < 0n ? -mismatchCents : mismatchCents;
      const direction: "TOPUP" | "WITHDRAW" = mismatchCents > 0n ? "TOPUP" : "WITHDRAW";
      // Positive mismatch (reserves > wallets) → user needs to TOPUP wallets.
      // Negative mismatch (wallets > reserves) → user needs to WITHDRAW excess.
      // Confirm sign convention by reading reserves-summary-builder.ts — adjust if inverted.

      const payload: ReserveTopupPayload = {
        shortfall_cents: absShortfall.toString(),
        direction,
        currency: summary.currency ?? summary.totals.currency,  // pick matching field from DTO
      };
      await deps.taskRepo.emitReserveTopup(
        input.tenantId,
        input.budgetId,
        payload,
        tx,
      );
    }
