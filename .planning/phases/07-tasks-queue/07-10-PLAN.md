    ```typescript
    import { Page, expect } from "@playwright/test";

    export class BdpPage {
      constructor(private page: Page) {}

      async expandTaskBanner(): Promise<void> {
        const banner = this.page.getByRole("region", { name: /tasks/i });
        // Phase 3 ARIA accordion — assumes header is clickable
        await banner.getByRole("button").first().click();
      }

      async assertTaskTitleVisible(title: string): Promise<void> {
        await expect(this.page.getByText(title)).toBeVisible();
      }

      async assertActionButtonLabel(label: string): Promise<void> {
        await expect(this.page.getByRole("button", { name: label })).toBeVisible();
      }

      async clickActionButton(): Promise<void> {
        // Click the first enabled action button inside a banner row
        const row = this.page.getByRole("listitem").first();
        await row.getByRole("button").last().click();
      }

      async assertTaskBannerEventuallyHidden(opts: { timeout: number }): Promise<void> {
        await expect(this.page.getByRole("region", { name: /tasks/i })).toBeHidden({ timeout: opts.timeout });
      }

      async assertTaskCount(count: number): Promise<void> {
        const rows = this.page.getByRole("listitem").filter({ hasText: /reserve|draft|cushion/i });
        await expect(rows).toHaveCount(count);
      }
    }
    ```
