    export function TaskBannerRow({ task, budgetId, onResolved }: TaskBannerRowProps) {
      const t = useTranslations();   // match existing useTranslations call site
      const router = useRouter();
      const [pending, setPending] = useState(false);

      const titleKey = `bdp.tasks.title.${task.kind}` as const;
      const kindKey = `bdp.tasks.kind.${task.kind}` as const;
      const actionKey = `bdp.tasks.action.${task.kind}.label` as const;
      const ariaLabelKey = `bdp.tasks.action.${task.kind}.ariaLabel` as const;

      // Title ICU placeholders — extract from payload, format via Intl.NumberFormat.
      const titleParams = buildTitleParams(task);

      async function handleAction() {
        switch (task.kind) {
          case "RESERVE_TOPUP":
            router.push(`/budgets/${budgetId}/reserves?task=${task.id}`);
            break;
          case "CUSHION_BELOW_TARGET":
            router.push(`/budgets/${budgetId}/wallets?task=${task.id}#cushion`);
            break;
          case "CONFIRM_DRAFT": {
            setPending(true);
            try {
              const draftId = task.payload?.draft_id as string | undefined;
              if (!draftId) throw new Error("Missing draft_id in CONFIRM_DRAFT payload");
              // Match the existing confirm-draft endpoint shape — verify against
              // apps/api/src/routes/recurring-rules.ts (or matching path) for the exact URL.
              const res = await clientApiFetch(
                `/recurring-rules/drafts/${draftId}/confirm`,
                { method: "POST", headers: { "X-Budget-ID": budgetId } },
              );
              if (!res.ok) throw new Error(`Confirm failed: ${res.status}`);
              // Optimistic row collapse — parent removes from list; backend auto-resolves on next poll
              onResolved?.(task.id);
            } catch (e) {
              console.error("[task-banner-row] confirm draft failed:", e);
              toast.error(t("bdp.tasks.confirmError"));
            } finally {
              setPending(false);
            }
            break;
          }
        }
      }

      const isDeepLink = task.kind === "RESERVE_TOPUP" || task.kind === "CUSHION_BELOW_TARGET";

      return (
        <div
          role="listitem"
          className="flex h-12 items-center gap-3 border-b border-[var(--hairline-dark)] bg-[var(--surface-card-dark)] px-4"
        >
          <span className="flex-1 truncate text-sm text-[var(--body-on-dark)]">
            {t(titleKey, titleParams)}
          </span>
          <Badge variant="secondary">{t(kindKey)}</Badge>
          <Button
            variant="primary"
            size="sm"
            onClick={handleAction}
            disabled={pending}
            aria-busy={pending || undefined}
            aria-label={isDeepLink ? t(ariaLabelKey) : undefined}
          >
            {pending ? (
              <Loader2 className="h-3.5 w-3.5 animate-spin" aria-hidden="true" />
            ) : (

    function buildTitleParams(task: TaskBannerRowProps["task"]): Record<string, string> {
      const payload = task.payload ?? {};
      const currency = (payload.currency as string) ?? "EUR";

      function fmt(cents: string | undefined): string {
        if (!cents) return "";
        return new Intl.NumberFormat(undefined, {
          style: "currency",
          currency,
        }).format(Number(cents) / 100);
      }

      switch (task.kind) {
        case "RESERVE_TOPUP":
          return { amount: fmt(payload.shortfall_cents as string) };
        case "CONFIRM_DRAFT":
          return {
            ruleName: (payload.rule_name as string) ?? "",
            amount: fmt(payload.amount_cents as string),
          };
        case "CUSHION_BELOW_TARGET":
          return { shortfall: fmt(payload.shortfall_cents as string) };
      }
    }
