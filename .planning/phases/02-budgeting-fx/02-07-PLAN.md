Add the **edit-via-correction-row** path on top of plan 02-06's transaction writer (D-01-a/b, EXPN-06). Editing inserts a NEW ledger row marked `corrects_id = original.id` with whatever fields changed (re-running FX lookup if amount/currency/date changed); the original row is untouched at SQL because UPDATE is REVOKE'd. Purpose: This is the only legitimate way to "fix" a past transaction in our append-only model. Plan 02-09's bulk re-categorize calls the same correction-row builder per row.

       ```typescript
       import { newUuid } from '@budget/shared-kernel';
       import type { TransactionRow } from './transaction';

       export interface CorrectionEdits {
         amountOrig?: string; currencyOrig?: string;
         transactionDate?: string; categoryId?: string | null;
         accountId?: string; note?: string | null;
         // For TRANSFER leg edits: edits apply only to this leg; transfer_group_id preserved.
         // FX result (re-computed in use case if amount/currency/date changed):
         amountDefault?: string; fxRate?: string; fxRateDate?: string; fxProvider?: string;
       }

       export function buildCorrectionRow(original: TransactionRow, edits: CorrectionEdits, actorUserId: string, now: Date): TransactionRow {
         return {
           ...original,
           id: newUuid(),
           correctsId: original.id,
           createdAt: now,
           actorUserId,
           // Apply only the explicitly-changed fields:
           amountOrig: edits.amountOrig ?? original.amountOrig,
           currencyOrig: edits.currencyOrig ?? original.currencyOrig,
           amountDefault: edits.amountDefault ?? original.amountDefault,
           currencyDefault: original.currencyDefault,  // immutable per workspace setting
           fxRate: edits.fxRate ?? original.fxRate,
           fxRateDate: edits.fxRateDate ?? original.fxRateDate,
           fxProvider: edits.fxProvider ?? original.fxProvider,
           transactionDate: edits.transactionDate ?? original.transactionDate,
           categoryId: edits.categoryId !== undefined ? edits.categoryId : original.categoryId,
           accountId: edits.accountId ?? original.accountId,
           note: edits.note !== undefined ? edits.note : original.note,
         };
       }

       export function computeDiff(original: TransactionRow, edits: CorrectionEdits): Record<string, {before: unknown; after: unknown}> {
         const diff: Record<string, {before: unknown; after: unknown}> = {};
         for (const [k, v] of Object.entries(edits)) {
           const o = (original as any)[k];
           if (o !== v) diff[k] = { before: o, after: v };
         }
         return diff;
       }
       ```
