       ```typescript
       export class DrizzleTransactionRepo implements TransactionRepo {
         constructor(private readonly accountRepo: AccountRepo, private readonly projectionRepo: SpendingProjectionRepo) {}
         async create(rows: TransactionRow[], userId: UserId, tenantId: TenantId): Promise<void> {
           const r = await withTenantTx(tenantId, userId, async (tx) => {
             for (const row of rows) {
               await tx.execute(sql`INSERT INTO budgeting.expense_ledger (...) VALUES (...)`);
               // Sync balance (D-05-e)
               const sign = row.kind === 'INCOME' ? 1 : -1;  // TRANSFER legs: from-leg debit (-), to-leg credit (+) — caller sets sign per row
               await this.accountRepo.applyDelta(tx, row.accountId, String(sign * Number(row.amountOrig)));
               // Sync projection (ENGR-14) — only for EXPENSE/INCOME with categoryId
               if (row.kind !== 'TRANSFER' && row.categoryId) {
                 await this.projectionRepo.upsert(tx, { tenantId: row.tenantId, workspaceId: row.tenantId /* workspace==tenant in this model */, categoryId: row.categoryId, monthStart: firstDayOfTransactionMonth, deltaNormal: row.kind === 'EXPENSE' ? row.amountDefault : '0', deltaCushion: '0', currency: row.currencyDefault });

               // Outbox (Pitfall 7)
               await writeOutbox(tx, { tenantId: row.tenantId, aggregateType: 'transaction', aggregateId: row.id, eventType: 'budgeting.transaction.created', payload: { ...row } });
             }
           });
           if (r.isErr()) throw r.error;
         }
       }
       ```
