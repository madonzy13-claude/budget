function resolveQuickEntryDate(
  month: Temporal.PlainYearMonth,
  today: Temporal.PlainDate,
): string {
  if (Temporal.PlainYearMonth.compare(month, today.toPlainYearMonth()) === 0)
    return today.toString();
  return month.toPlainDate({ day: month.daysInMonth }).toString();

## Trust Boundaries

| Boundary                     | Description                                                                    |
| ---------------------------- | ------------------------------------------------------------------------------ |
| RSC searchParams `?month`    | regex-validated `/^\d{4}-\d{2}$/`; falls back to current month on malformed    |
| RSC → API serverApiFetch     | budgetId passed as first arg sets X-Budget-ID header (T-03-06-08 from Phase 3) |
| slider form input → mutation | Zod schema validates body before POST/PATCH                                    |
| dnd-kit drag listeners       | scoped to grip handle only; touch-action:none prevents scroll conflict         |

## STRIDE Threat Register

| Threat ID  | Category               | Component                                                 | Disposition | Mitigation Plan                                                                                                                                                           |
| ---------- | ---------------------- | --------------------------------------------------------- | ----------- | ------------------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| T-04-04-01 | Tampering              | Open redirect via `?month`                                | mitigate    | Regex `/^\d{4}-\d{2}$/` validation; never used in a redirect URL (RESEARCH §Security row 9)                                                                               |
| T-04-04-02 | Information Disclosure | Slider leaks other tenant's categories in dropdown        | mitigate    | categories list comes from RSC `/budgets/:id/categories` which is RLS-scoped; client never fetches cross-budget                                                           |
| T-04-04-03 | Tampering              | Drag-reorder + quick-entry cache race                     | mitigate    | Separate query keys per Pitfall 2: `["categories", budgetId]` vs `["transactions", budgetId, month]` vs `["drafts", budgetId, month]`; `cancelQueries()` in each onMutate |
| T-04-04-04 | Denial of Service      | Mobile drag conflicts with horizontal scroll (user stuck) | mitigate    | `touch-action: none` ONLY on grip handle; TouchSensor `delay: 200, tolerance: 8`; Pitfall 1 mitigations                                                                   |
| T-04-04-05 | Information Disclosure | TransactionSlider exposes FX rate as-of in cleartext      | accept      | FX as-of is intentionally surfaced per TXN-06 / D-PH4-S2; no PII                                                                                                          |
| T-04-04-06 | Spoofing               | XSS via category name in slider dropdown / row label      | mitigate    | React JSX auto-escapes text by default; Zod validation on category name rejects HTML control chars; never use raw HTML injection sinks for user-provided strings          |
| T-04-04-07 | Tampering              | RSC server-side fetch leaks session to wrong budget       | mitigate    | every `serverApiFetch` call passes `id` as first arg → X-Budget-ID header; verified in layout.tsx pattern                                                                 |
| T-04-04-08 | Repudiation            | slider Delete bypasses confirmation                       | mitigate    | AlertDialog with grid.confirm.deleteTxn.\* required before DELETE mutate fires                                                                                            |

3. **`spendings-grid-client.tsx`** — per RESEARCH §Pattern 1 (lines 511-549) + §Pattern 5 (no-hover regression) + UI-SPEC §1. **Critical:** transactionsByCatId and draftsByCatId are derived from useTransactions / useDrafts hook results via useMemo (NOT from props directly — props seed initialData):

   ```typescript
   "use client";
   import { useState, useMemo } from "react";
   import {
     DndContext, PointerSensor, KeyboardSensor, TouchSensor, useSensor, useSensors,
     closestCenter, type DragEndEvent,
   } from "@dnd-kit/core";
   import {
     SortableContext, sortableKeyboardCoordinates, horizontalListSortingStrategy, arrayMove,
   } from "@dnd-kit/sortable";
   import { Temporal } from "temporal-polyfill";
   import { CategoryColumn } from "./category-column";
   import { AddCategoryColumn } from "./add-category-column";
   import { MonthNavigator } from "./month-navigator";
   import { TransactionSlider } from "../transaction-slider";
   import { CategorySlider } from "../category-slider";
   import { useReorderCategories } from "@/hooks/use-reorder-categories";
   import { useMonthParam } from "@/hooks/use-month-param";
   import { useSpendingsSummary } from "@/hooks/use-spendings-summary";
   import { useTransactions, type TxnDTO } from "@/hooks/use-transactions";
   import { useDrafts, type DraftDTO } from "@/hooks/use-drafts";

   export function SpendingsGridClient(props: SpendingsGridClientProps) {
     const { budgetId, budgetCurrency, budgetTz } = props;
     const { monthStr, isCurrentMonth } = useMonthParam(budgetTz);
     const month = monthStr;

     // Query hooks: hydrate from RSC initialData; live data after first refetch.
     // queryKey contract: must match mutation hooks' invalidate keys (see Plan 04-03 hooks).
     const summary = useSpendingsSummary(budgetId, month, props.initialSummary);
     const txns = useTransactions(budgetId, month, { initialData: props.initialTransactions });
     const drafts = useDrafts(budgetId, month, { initialData: props.initialDrafts });

     const [localCategoryOrder, setLocalCategoryOrder] = useState(props.initialCategories);
     const reorder = useReorderCategories(budgetId);
     const sensors = useSensors(
       useSensor(PointerSensor, { activationConstraint: { distance: 4 } }),
       useSensor(TouchSensor, { activationConstraint: { delay: 200, tolerance: 8 } }),
       useSensor(KeyboardSensor, { coordinateGetter: sortableKeyboardCoordinates }),
     );

     const [txSlider, setTxSlider] = useState<{ open: boolean; mode: "create" | "edit"; txId?: string; prefillCategoryId?: string }>({ open: false, mode: "create" });
     const [catSlider, setCatSlider] = useState<{ open: boolean; mode: "create" | "edit"; categoryId?: string }>({ open: false, mode: "create" });

     function handleDragEnd(e: DragEndEvent) {
       const { active, over } = e;
       if (!over || active.id === over.id) return;
       const oldIndex = localCategoryOrder.findIndex(c => c.id === active.id);
       const newIndex = localCategoryOrder.findIndex(c => c.id === over.id);
       const newOrder = arrayMove(localCategoryOrder, oldIndex, newIndex);
       const prev = localCategoryOrder;
       setLocalCategoryOrder(newOrder);
       reorder.mutate({ orderedIds: newOrder.map(c => c.id) }, {
         onError: () => setLocalCategoryOrder(prev),
       });
     }

     const today = Temporal.Now.plainDateISO(budgetTz);
     const ym = Temporal.PlainYearMonth.from(month);
     const resolvedQuickEntryDate = isCurrentMonth
       ? today.toString()
       : ym.toPlainDate({ day: ym.daysInMonth }).toString();

     // ---- Per-category Maps derived from query hook data (BLOCKER FIX) ----
     // These Maps MUST come from useTransactions/useDrafts hook .data — NOT from props directly,
     // because the hooks reflect live cache state after optimistic mutations + revalidates.
     // Hydration: hook initialData seeded by props.initialTransactions / props.initialDrafts,
     // so first render produces the same Maps as a pure prop-derivation would.
     const summaryByCatId = useMemo(
       () => new Map(summary.data?.categories.map(c => [c.categoryId, c]) ?? []),
       [summary.data],
     );
     const transactionsByCatId = useMemo(() => {
       const m = new Map<string, TxnDTO[]>();
       for (const t of txns.data ?? []) {
         const list = m.get(t.categoryId) ?? [];
         list.push(t);
         m.set(t.categoryId, list);
       }
       // Newest first (UI-SPEC §3 — txn list ordering)
       for (const list of m.values()) {
         list.sort((a, b) => b.transactionDate.localeCompare(a.transactionDate));
       }
       return m;
     }, [txns.data]);
     const draftsByCatId = useMemo(() => {
       const m = new Map<string, DraftDTO[]>();
       for (const d of drafts.data ?? []) {
         const list = m.get(d.categoryId) ?? [];
         list.push(d);
         m.set(d.categoryId, list);
       }
       return m;
     }, [drafts.data]);
     // ---- end Maps derivation ----

     return (
       <>
         <MonthNavigator month={month} budgetTz={budgetTz} />
         <div data-testid="spendings-grid" className="overflow-x-auto px-[var(--spacing-xl)] sm:px-[var(--spacing-md)]">
           <DndContext sensors={sensors} collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
             <SortableContext items={localCategoryOrder.map(c => c.id)} strategy={horizontalListSortingStrategy}>
               <div className="flex gap-[var(--spacing-xxs)]">
                 {localCategoryOrder.map(c => (
                   <CategoryColumn
                     key={c.id}
                     category={c}
                     summary={summaryByCatId.get(c.id) ?? defaultEmptySummary(c.id)}
                     cushionModeEnabled={summary.data?.cushionModeEnabled ?? false}
                     budgetCurrency={budgetCurrency}
                     transactions={transactionsByCatId.get(c.id) ?? []}
                     drafts={draftsByCatId.get(c.id) ?? []}
                     budgetId={budgetId}
                     month={month}
                     isPastMonth={!isCurrentMonth}
                     resolvedQuickEntryDate={resolvedQuickEntryDate}
                     onEditTxn={(txId) => setTxSlider({ open: true, mode: "edit", txId })}
                     onEditDraft={(draftId) => setTxSlider({ open: true, mode: "edit", txId: draftId })}
                     onEditCategory={(categoryId) => setCatSlider({ open: true, mode: "edit", categoryId })}
                   />
                 ))}
               </div>
             </SortableContext>
             {/* AddCategoryColumn is a sibling of SortableContext children, NOT inside the items list (D-PH4-D4) */}
             <AddCategoryColumn onClick={() => setCatSlider({ open: true, mode: "create" })} />
           </DndContext>
         </div>
         <TransactionSlider
           open={txSlider.open}
           onOpenChange={(o) => setTxSlider({ ...txSlider, open: o })}
           mode={txSlider.mode}
           budgetId={budgetId}
           month={month}
           budgetCurrency={budgetCurrency}
           categories={localCategoryOrder}
         />
         <CategorySlider
           open={catSlider.open}
           onOpenChange={(o) => setCatSlider({ ...catSlider, open: o })}
           mode={catSlider.mode}
           budgetId={budgetId}
           budgetCurrency={budgetCurrency}
         />
       </>
     );
   }
   ```

   Note: AddCategoryColumn is rendered OUTSIDE the SortableContext's items list (the wrapping `<div className="flex gap-...">` that contains the sortable CategoryColumns). It is a sibling within DndContext but not registered as a sortable item, so it cannot be dragged or used as a drop target.    import { getTranslations } from "next-intl/server";
   import { serverApiFetch } from "@/lib/budget-fetch.server";
   import { SpendingsGridClient } from "@/components/budgeting/spendings-grid/spendings-grid-client";
   import { Temporal } from "temporal-polyfill";

   interface PageProps {
     params: Promise<{ locale: string; id: string }>;
     searchParams: Promise<{ month?: string }>;
   }

   export default async function SpendingsPage({ params, searchParams }: PageProps) {
     const { locale, id: budgetId } = await params;
     const { month: monthParam } = await searchParams;
     const month = monthParam && /^\d{4}-\d{2}$/.test(monthParam)
       ? monthParam
       : Temporal.Now.plainDateISO().toPlainYearMonth().toString();

     const [categoriesRes, txnsRes, draftsRes, summaryRes] = await Promise.all([
       serverApiFetch(budgetId, `/budgets/${budgetId}/categories`),
       serverApiFetch(budgetId, `/budgets/${budgetId}/transactions?month=${month}&confirmed=true`),
       serverApiFetch(budgetId, `/budgets/${budgetId}/transactions?month=${month}&confirmed=false`),
       serverApiFetch(budgetId, `/budgets/${budgetId}/spendings-summary?month=${month}`),
     ]);

     const categories = categoriesRes.ok ? (await categoriesRes.json()).categories : [];
     const transactions = txnsRes.ok ? (await txnsRes.json()).transactions : [];
     const drafts = draftsRes.ok ? (await draftsRes.json()).transactions : [];
     const summary = summaryRes.ok
       ? await summaryRes.json()
       : { categories: [], cushionModeEnabled: false, budgetCurrency: "USD", budgetTz: "UTC", month };

     return (
       <SpendingsGridClient
         budgetId={budgetId}
         budgetCurrency={summary.budgetCurrency ?? "USD"}
         budgetTz={summary.budgetTz ?? "UTC"}
         month={month}
         initialCategories={categories}
         initialTransactions={transactions}
         initialDrafts={drafts}
         initialSummary={summary}
       />
     );
   }
