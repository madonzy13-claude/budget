    ```ts
    import { ok, err, type Result } from "@budget/shared-kernel";
    import type { BudgetHomeSummaryRepo } from "../ports/budget-home-summary-repo";
    import type { FxProvider } from "../ports/fx-provider";

    export interface GetBudgetHomeSummaryInput {
      tenantId: string;
      budgetId: string;
      userId: string;
      now: Date;
    }

    export interface HomeSummaryDTO { /* shape from <interfaces> above */ }

    export interface GetBudgetHomeSummaryDeps {
      summaryRepo: BudgetHomeSummaryRepo;
      fxProvider: FxProvider;
    }

    export function getBudgetHomeSummary(deps: GetBudgetHomeSummaryDeps) {
      return async (input: GetBudgetHomeSummaryInput): Promise<Result<HomeSummaryDTO, Error>> => {
        try {
          const meta = await deps.summaryRepo.getBudgetMeta(input.budgetId, input.tenantId);
          if (!meta) return err(new Error("budget_not_found"));

          const userDisplay = await deps.summaryRepo.getUserDisplayCurrency(input.userId);
          const displayCurrency = userDisplay && userDisplay.trim().length === 3 ? userDisplay : meta.default_currency;

          // Month boundary using Temporal-correct boundaries via input.now (UTC).
          const monthStart = new Date(Date.UTC(input.now.getUTCFullYear(), input.now.getUTCMonth(), 1));
          const monthEnd = new Date(Date.UTC(input.now.getUTCFullYear(), input.now.getUTCMonth() + 1, 1));

          const [spentCents, wallets, overspent] = await Promise.all([
            deps.summaryRepo.sumCurrentMonthSpend(input.budgetId, input.tenantId, monthStart, monthEnd),
            deps.summaryRepo.listWalletsForBudget(input.budgetId, input.tenantId),
            deps.summaryRepo.topOverspentCategories(input.budgetId, input.tenantId, monthStart, monthEnd, meta.cushion_mode_enabled, 2),
          ]);

          // FX-convert each wallet to displayCurrency and sum
          let walletsSumCents = 0n;
          const convertedAt = new Date();
          for (const w of wallets) {
            const converted = await deps.fxProvider.convert(
              { amountCents: w.amount_cents, currency: w.currency },
              displayCurrency,
              convertedAt,
            );
            walletsSumCents += converted.amountCents;
          }

          return ok({
            budgetId: input.budgetId,
            name: meta.name,
            kind: meta.kind,
            default_currency: meta.default_currency,
            display_currency: displayCurrency,
            spent_current_month: { amount_cents: spentCents.toString(), currency: meta.default_currency },
            wallets_value_display_ccy: { amount_cents: walletsSumCents.toString(), currency: displayCurrency, converted_at: convertedAt.toISOString() },
            top_overspent: overspent.map((o) => ({ category_id: o.category_id, category_name: o.category_name, over_amount_cents: o.over_amount_cents.toString() })),
          });
        } catch (e) { return err(e as Error); }
      };
    }
    ```

    ```ts
    import { sql } from "drizzle-orm";
    import { withTenantTx } from "@budget/platform";
    import { TenantId, UserId } from "@budget/shared-kernel";
    import type { BudgetHomeSummaryRepo } from "../../ports/budget-home-summary-repo";

    const SYSTEM_USER_ID = "00000000-0000-0000-0000-000000000001";

    export function createBudgetHomeSummaryRepo(): BudgetHomeSummaryRepo {
      return {
        async getBudgetMeta(budgetId, tenantId) {
          const r = await withTenantTx(TenantId(tenantId), UserId(SYSTEM_USER_ID), async (tx) => {
            const drizzleTx = tx as { execute: (q: unknown) => Promise<{ rows: any[] }> };
            const res = await drizzleTx.execute(sql`
              SELECT name, kind, default_currency, COALESCE(cushion_mode_enabled, false) AS cushion_mode_enabled
                FROM tenancy.budgets
               WHERE id = ${budgetId}::uuid AND tenant_id = ${tenantId}::uuid AND deleted_at IS NULL
               LIMIT 1
            `);
            return res.rows[0] ?? null;
          });
          if (r.isErr()) throw r.error;
          return r.value as any;
        },

        async sumCurrentMonthSpend(budgetId, tenantId, monthStart, monthEnd) {
          const r = await withTenantTx(TenantId(tenantId), UserId(SYSTEM_USER_ID), async (tx) => {
            const drizzleTx = tx as { execute: (q: unknown) => Promise<{ rows: any[] }> };
            const res = await drizzleTx.execute(sql`
              SELECT COALESCE(SUM(amount_converted_cents), 0)::text AS total
                FROM budgeting.transactions
               WHERE budget_id = ${budgetId}::uuid
                 AND tenant_id = ${tenantId}::uuid
                 AND transaction_date >= ${monthStart.toISOString()}
                 AND transaction_date <  ${monthEnd.toISOString()}
                 AND confirmed_at IS NOT NULL
                 AND deleted_at IS NULL
            `);
            return BigInt(res.rows[0]?.total ?? 0);
          });
          if (r.isErr()) throw r.error;
          return r.value;
        },

        async listWalletsForBudget(budgetId, tenantId) {
          const r = await withTenantTx(TenantId(tenantId), UserId(SYSTEM_USER_ID), async (tx) => {
            const drizzleTx = tx as { execute: (q: unknown) => Promise<{ rows: any[] }> };
            const res = await drizzleTx.execute(sql`
              SELECT amount_cents::text, currency
                FROM tenancy.wallets
               WHERE budget_id = ${budgetId}::uuid
                 AND tenant_id = ${tenantId}::uuid
                 AND deleted_at IS NULL
            `);
            return res.rows.map((row) => ({ amount_cents: BigInt(row.amount_cents), currency: row.currency as string }));
          });
          if (r.isErr()) throw r.error;
          return r.value;
        },

        async topOverspentCategories(budgetId, tenantId, monthStart, monthEnd, useCushion, limit) {
          // Compute over = max(0, spent - active_budget) per category for the current month.
          // active_budget = cushion_amount_cents if useCushion else planned_amount_cents,
          // from category_limits SCD-2 row active during [monthStart, monthEnd).
          const r = await withTenantTx(TenantId(tenantId), UserId(SYSTEM_USER_ID), async (tx) => {
            const drizzleTx = tx as { execute: (q: unknown) => Promise<{ rows: any[] }> };
            const res = await drizzleTx.execute(sql`
              WITH spent AS (
                SELECT category_id, COALESCE(SUM(amount_converted_cents), 0) AS spent_cents
                  FROM budgeting.transactions
                 WHERE budget_id = ${budgetId}::uuid
                   AND tenant_id = ${tenantId}::uuid
                   AND transaction_date >= ${monthStart.toISOString()}
                   AND transaction_date <  ${monthEnd.toISOString()}
                   AND deleted_at IS NULL
                 GROUP BY category_id
              ),
              limits AS (
                SELECT cl.category_id,
                       CASE WHEN ${useCushion} THEN cl.cushion_amount_cents ELSE cl.planned_amount_cents END AS active_budget_cents
                  FROM budgeting.category_limits cl
                 WHERE cl.budget_id = ${budgetId}::uuid
                   AND cl.tenant_id = ${tenantId}::uuid
                   AND cl.effective_from <= ${monthStart.toISOString()}
                   AND (cl.effective_to IS NULL OR cl.effective_to > ${monthStart.toISOString()})
              )
              SELECT s.category_id, c.name AS category_name,
                     GREATEST(0, s.spent_cents - COALESCE(l.active_budget_cents, 0))::text AS over_amount_cents
                FROM spent s
                LEFT JOIN limits l ON l.category_id = s.category_id
                JOIN budgeting.categories c ON c.id = s.category_id
               WHERE GREATEST(0, s.spent_cents - COALESCE(l.active_budget_cents, 0)) > 0
               ORDER BY over_amount_cents DESC
               LIMIT ${limit}
            `);
            return res.rows.map((row) => ({
              category_id: row.category_id as string,
              category_name: row.category_name as string,
              over_amount_cents: BigInt(row.over_amount_cents),
            }));
          });
          if (r.isErr()) throw r.error;
          return r.value;
        },

        async getUserDisplayCurrency(userId) {
          // identity.users is OUTSIDE per-tenant RLS — use withInfraTx OR direct pool. Reserve-balance pattern uses withTenantTx for tenant tables. For user lookup, use existing pattern in apps/api/src/routes/settings.ts (already reads users.display_currency).
          const r = await withTenantTx(TenantId("00000000-0000-0000-0000-000000000000"), UserId(userId), async (tx) => {
            const drizzleTx = tx as { execute: (q: unknown) => Promise<{ rows: any[] }> };
            const res = await drizzleTx.execute(sql`
              SELECT display_currency FROM identity.users WHERE id = ${userId}::uuid LIMIT 1
            `);
            return (res.rows[0]?.display_currency as string | undefined) ?? "";
          });
          if (r.isErr()) return "";
          return r.value;
        },
      };
    }
    ```
