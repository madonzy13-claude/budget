**Implications for this plan's SQL:**

1. Read `tenancy.budgets` by `WHERE id = ${budgetId}::uuid` (do NOT filter by `tenant_id` — that column does not exist on this table). No `deleted_at IS NULL` filter.
2. Read `budgeting.wallets` by `WHERE tenant_id = ${budgetId}::uuid` (budget_id === tenant_id). Column is `current_balance numeric(19,4)`, NOT `amount_cents bigint`. Convert to cents at adapter boundary: `(current_balance * 100)::bigint`.
3. Read `budgeting.expense_ledger` (NOT `budgeting.transactions`) by `WHERE budget_id = ${budgetId}::uuid AND tenant_id = ${budgetId}::uuid`. Date column is `transaction_date`. Sum `amount_converted_cents`.
4. Read `budgeting.category_limits` by `WHERE tenant_id = ${budgetId}::uuid` (no budget_id column). Use `cushion_amount` not `cushion_amount_cents`.
5. Read `budgeting.categories` by `WHERE tenant_id = ${budgetId}::uuid` (no budget_id column).
6. Read `identity.users.display_currency` via `deps.identity.userRepo.findById(UserId(userId))` — DO NOT inject a raw withTenantTx with zero-uuid (identity is cross-tenant; that pattern is forbidden).

      budgetId: string;      // == tenantId per v1.1 invariant
      userId: string;
      now: Date;
    }

    export interface HomeSummaryDTO { /* shape from <interfaces> above */ }

    export interface GetBudgetHomeSummaryDeps {
      summaryRepo: BudgetHomeSummaryRepo;
      fxProvider: FxProvider;
      userRepo: UserRepo;
    }

    export function getBudgetHomeSummary(deps: GetBudgetHomeSummaryDeps) {
      return async (input: GetBudgetHomeSummaryInput): Promise<Result<HomeSummaryDTO, Error>> => {
        try {
          const meta = await deps.summaryRepo.getBudgetMeta(input.budgetId);
          if (!meta) return err(new Error("budget_not_found"));

          // Identity is cross-tenant. Use the existing UserRepo.findById which internally uses
          // withUserContext (sets app.current_user_id GUC, NOT app.tenant_ids).
          const user = await deps.userRepo.findById(UserId(input.userId));
          const userDisplay = user?.display_currency ?? "";
          const displayCurrency = userDisplay && userDisplay.trim().length === 3 ? userDisplay : meta.default_currency;

          // UTC month boundaries
          const monthStart = new Date(Date.UTC(input.now.getUTCFullYear(), input.now.getUTCMonth(), 1));
          const monthEnd = new Date(Date.UTC(input.now.getUTCFullYear(), input.now.getUTCMonth() + 1, 1));

          const [spentCents, wallets, overspent] = await Promise.all([
            deps.summaryRepo.sumCurrentMonthSpend(input.budgetId, monthStart, monthEnd),
            deps.summaryRepo.listWalletsForBudget(input.budgetId),
            deps.summaryRepo.topOverspentCategories(input.budgetId, monthStart, monthEnd, meta.cushion_mode_enabled, 2),
          ]);

          // FX-convert each wallet to displayCurrency and sum
          let walletsSumCents = 0n;
          const convertedAt = new Date();
          for (const w of wallets) {
            const converted = await deps.fxProvider.convert(
              { amountCents: w.amount_cents, currency: w.currency },
              displayCurrency,
              convertedAt,
            );
            walletsSumCents += converted.amountCents;
          }

          return ok({
            budgetId: input.budgetId,
            name: meta.name,
            kind: meta.kind,
            default_currency: meta.default_currency,
            display_currency: displayCurrency,
            spent_current_month: { amount_cents: spentCents.toString(), currency: meta.default_currency },
            wallets_value_display_ccy: { amount_cents: walletsSumCents.toString(), currency: displayCurrency, converted_at: convertedAt.toISOString() },
            top_overspent: overspent.map((o) => ({ category_id: o.category_id, category_name: o.category_name, over_amount_cents: o.over_amount_cents.toString() })),
          });
        } catch (e) { return err(e as Error); }
      };
    }

            }));
          });
