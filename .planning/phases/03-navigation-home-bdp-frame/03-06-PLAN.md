    ```tsx
    "use client";
    import Link from "next/link";
    import { usePathname } from "next/navigation";
    import { useTranslations } from "next-intl";
    import { LayoutGrid, Coins, Wallet, Settings, type LucideIcon } from "lucide-react";
    import { cn } from "@/lib/utils";

    interface BdpTabsProps { locale: string; budgetId: string; }

    const TABS: Array<{ slug: "spendings" | "reserves" | "wallets" | "settings"; icon: LucideIcon }> = [
      { slug: "spendings", icon: LayoutGrid },
      { slug: "reserves", icon: Coins },
      { slug: "wallets", icon: Wallet },
      { slug: "settings", icon: Settings },
    ];

    export function BdpTabs({ locale, budgetId }: BdpTabsProps) {
      const pathname = usePathname() ?? "";
      const t = useTranslations("bdp.tab");
      return (
        <nav aria-label={t("aria")} className="flex gap-2 px-4 sm:px-6 h-12 items-center">
          {TABS.map(({ slug, icon: Icon }) => {
            const href = `/${locale}/budgets/${budgetId}/${slug}`;
            const active = pathname.startsWith(href);
            const label = t(`${slug}.label`);
            return (
              <Link
                key={slug}
                href={href}
                aria-current={active ? "page" : undefined}
                aria-label={label}
                className={cn(
                  "inline-flex items-center gap-2 h-9 rounded-[var(--radius-pill)] px-4 transition-colors focus-visible:outline focus-visible:outline-2 focus-visible:outline-[var(--info-ring)] focus-visible:outline-offset-2",
                  "min-w-[44px] min-h-[44px] sm:min-w-0 sm:min-h-0",
                  active
                    ? "bg-[var(--primary)] text-[var(--on-primary)] text-title-sm font-semibold"
                    : "text-[var(--muted-foreground)] hover:bg-[var(--surface-elevated-dark)] hover:text-[var(--body-on-dark)]",
                )}
              >
                <Icon className="size-[18px]" aria-hidden="true" />
                <span className={cn(active ? "inline" : "hidden sm:inline")}>{label}</span>
              </Link>
            );
          })}
        </nav>
      );
    }
    ```
