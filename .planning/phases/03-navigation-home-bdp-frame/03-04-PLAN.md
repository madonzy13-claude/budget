<pathname_header_decision>
## BLOCKER #7 resolution — middleware-injected `x-pathname` request header (Option A)

Next.js App Router does NOT expose the current request pathname inside RSC layouts via the built-in `headers()` API — none of `x-next-pathname`, `x-pathname`, `x-invoke-path` are documented public surfaces. The reliable pattern is to inject a custom header from middleware (which runs on every request and DOES know the URL).

**Decision:** extend the existing `apps/web/src/middleware.ts` to set `x-pathname` on the forwarded request. The (app) layout reads it via `headers()` and matches `/budgets/{id}` to derive `activeBudgetId`.

Reference: https://nextjs.org/docs/app/api-reference/file-conventions/middleware — the recommended pattern for "passing data from middleware to server components" is `NextResponse.next({ request: { headers } })` with a mutated `Headers` clone.

The fallback path (BudgetSwitcher renders without an active row when header is absent) is graceful — this is purely a "highlight the active budget" affordance, not a security boundary.
</pathname_header_decision>

    "use client";
    import { useState, useCallback } from "react";
    import { useRouter } from "next/navigation";
    import { useTranslations } from "next-intl";
    import { Check, ChevronDown, Lock, Users, Plus } from "lucide-react";
    import { Popover, PopoverTrigger, PopoverContent } from "@/components/ui/popover";
    import { Badge } from "@/components/ui/badge";
    import { cn } from "@/lib/utils";

    export interface BudgetSummary {
      id: string;
      name: string;
      kind: "PRIVATE" | "SHARED";
      default_currency: string;
    }

    export interface BudgetSwitcherProps {
      budgets: BudgetSummary[];
      activeBudgetId: string | null;
      locale: string;
    }

    export function BudgetSwitcher({ budgets, activeBudgetId, locale }: BudgetSwitcherProps) {
      const t = useTranslations();
      const router = useRouter();
      const [open, setOpen] = useState(false);

      const active = budgets.find((b) => b.id === activeBudgetId) ?? null;
      const privateB = budgets.filter((b) => b.kind === "PRIVATE");
      const sharedB = budgets.filter((b) => b.kind === "SHARED");
      const isEmpty = budgets.length === 0;

      const onPick = useCallback((id: string) => {
        setOpen(false);
        if (id === activeBudgetId) return;
        router.push(`/${locale}/budgets/${id}/spendings`);
      }, [router, locale, activeBudgetId]);

      // Trigger label: active budget name OR empty hint
      const triggerLabel = isEmpty
        ? t("nav.switcher.empty.trigger")     // e.g. "No budgets yet"
        : active?.name ?? t("nav.switcher.empty.trigger");

      return (
        <Popover open={open} onOpenChange={setOpen}>
          <PopoverTrigger asChild>
            <button aria-label={t("nav.budgetSwitcher.trigger.aria")} className={/* UI-SPEC §2 trigger classes */ ""}>
              {!isEmpty && active && (active.kind === "PRIVATE" ? <Lock className="size-4" /> : <Users className="size-4" />)}
              <span className="text-title-sm max-w-[24ch] truncate sm:max-w-[12ch]">{triggerLabel}</span>
              <ChevronDown className="size-4" />
            </button>
          </PopoverTrigger>
          {/* z-[60] sits ABOVE the sticky top-nav header (z-50) and any BDP sticky wrapper (z-40). */}
          <PopoverContent className="z-[60] min-w-[256px] p-0">
            {isEmpty ? (
              <div className="p-4 space-y-3">
                <p className="text-sm text-[var(--muted-foreground)]">{t("nav.switcher.empty.body")}</p>
                <button
                  type="button"
                  onClick={() => { setOpen(false); router.push(`/${locale}/budgets/new`); }}
                  className="inline-flex items-center gap-2 text-sm text-[var(--primary)]"
                >
                  <Plus className="size-4" aria-hidden="true" />
                  {t("nav.switcher.empty.cta")}
                </button>
              </div>
            ) : (
              <>
                {privateB.length > 0 && renderGroup(privateB, t("nav.switcher.personal"), onPick, activeBudgetId)}
                {privateB.length > 0 && sharedB.length > 0 && <div className="h-px bg-[var(--hairline-dark)]" />}
                {sharedB.length > 0 && renderGroup(sharedB, t("nav.switcher.shared"), onPick, activeBudgetId)}
              </>
            )}
          </PopoverContent>
        </Popover>
      );
    }

    function renderGroup(rows: BudgetSummary[], heading: string, onPick: (id: string) => void, activeId: string | null) {
      return (
        <div className="py-2">
          <div className="px-4 pb-1 text-caption text-[var(--muted-foreground)]">{heading}</div>
          {rows.map((b) => {
            const isActive = b.id === activeId;
            const Icon = b.kind === "PRIVATE" ? Lock : Users;
            return (
              <button
                key={b.id}
                type="button"
                role="menuitemradio"
                aria-checked={isActive}
                onClick={() => onPick(b.id)}
                className={cn("flex h-10 w-full items-center gap-2 px-4 hover:bg-[var(--surface-elevated-dark)]")}
              >
                <span className="size-4">{isActive && <Check className="size-4" />}</span>
                <Icon className="size-4" aria-hidden="true" />
                <span className="flex-1 truncate text-left text-body-md">{b.name}</span>
                <Badge variant="outline" className="num text-[11px]">{b.default_currency}</Badge>
              </button>
            );
          })}
        </div>
      );
    }

    import { LocaleSelect } from "@/components/common/locale-select";

    interface TopNavProps {

    async function fetchBudgets(): Promise<BudgetSummary[]> {
      const res = await serverApiFetch(null, "/budgets/active");

          </div>
